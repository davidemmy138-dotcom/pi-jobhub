app.post('/approve', async (req, res) => { const { paymentId } = req.body; // Send approve request to Pi Server const response = await fetch(`https://api.minepi.com/v2/payments/${paymentId}/approve`, { method: "POST", headers: { "Authorization": `Key ${process.env.PI_API_KEY}`, // Your Pi API key "Content-Type": "application/json" } }); if (response.ok) { res.json({ success: true }); } else { res.status(400).json({ success: false }); } });
