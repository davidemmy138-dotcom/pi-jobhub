Frontend Implementation

To implement Pi payments in your frontend, you need to:

  1. Include the Pi SDK in your HTML
  2. Initialize the Pi SDK with your app credentials
  3. Create a payment function
  4. Handle the payment callback
<!-- Include Pi SDK in your HTML -->
<script src="https://sdk.minepi.com/pi-sdk.js"></script>

<script>
// Initialize the Pi SDK
const pi = new Pi({
  version: "2.0",
  sandbox: true // Set to false for production
});

// Create a payment function
async function createPayment(amount, memo) {
  const paymentData = {
    amount: amount,
    memo: memo,
    metadata: {
      productId: "demo_product"
    }
  };

  try {
    const payment = await pi.createPayment(paymentData);
    await pi.waitForPayment(payment.identifier);
    console.log("Payment completed:", payment);
    return payment;
  } catch (error) {
    console.error("Payment error:", error);
    throw error;
  }
}
</script>

Backend Implementation (Node.js)

Your backend needs to:

  1. Validate incoming payments
  2. Verify payment authenticity with Pi Servers
  3. Complete the payment process
  4. Handle webhooks from Pi Network
// server.js - Basic Pi payment handler
const express = require('express');
const axios = require('axios');
const app = express();

app.use(express.json());

// Endpoint to handle completed payments
app.post('/payments/complete', async (req, res) => {
  try {
    const { paymentId } = req.body;
    
    // Verify payment with Pi Server
    const response = await axios.get(`https://api.minepi.com/v2/payments/${paymentId}`, {
      headers: {
        Authorization: `Key ${process.env.PI_API_KEY}`
      }
    });
    
    const payment = response.data;
    
    // Check if payment is completed
    if (payment.status === 'completed') {
      // Fulfill the order
      console.log('Payment completed:', payment);
      res.json({ success: true, payment });
    } else {
      res.status(400).json({ error: 'Payment not completed' });
    }
  } catch (error) {
    console.error('Payment verification error:', error);
    res.status(500).json({ error: 'Payment verification failed' });
  }
});

// Start server
const PORT = process.env.PORT || 3000;
app.listen(PORT, () => {
  console.log(`Server running on port ${PORT}`);
});
// package.json - Dependencies
{
  "name": "pi-app-backend",
  "version": "1.0.0",
  "dependencies": {
    "express": "^4.18.2",
    "axios": "^1.4.0"
  }
}

Testing & Debugging

To test your Pi application:

  1. Use the Pi Testnet for development
  2. Test with the Pi Browser app
  3. Check console logs for errors
  4. Verify payment callbacks are working
  5. Test on different devices
// Example test script
const Pi = require('pi-backend-sdk');

// Initialize with testnet
const pi = new Pi({
  apiKey: process.env.PI_API_KEY,
  baseURL: 'https://api.testnet.minepi.com' // Testnet URL
});

// Test payment creation
async function testPayment() {
  try {
    const payment = await pi.createPayment({
      amount: 1,
      memo: 'Test payment',
      metadata: { productId: 'test' }
    });
    console.log('Test payment created:', payment);
    return payment;
  } catch (error) {
    console.error('Test payment failed:', error);
    throw error;
  }
}

testPayment();

Common Issues & Solutions

  • CORS errors: Ensure your server allows requests from your Pi app domain
  • Authentication failures: Verify your API key and app credentials
  • Payment not completing: Check that you're calling the completePayment method
  • Network issues: Test with different network connections
